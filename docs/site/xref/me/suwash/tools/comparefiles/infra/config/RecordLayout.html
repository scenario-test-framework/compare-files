1   package me.suwash.tools.comparefiles.infra.config;
2   
3   import java.util.ArrayList;
4   import java.util.LinkedHashMap;
5   import java.util.List;
6   import java.util.Map;
7   
8   import lombok.Getter;
9   import lombok.Setter;
10  import me.suwash.tools.comparefiles.infra.classification.RecordType;
11  import me.suwash.util.CompareUtils.CompareCriteria;
12  
13  /**
14   * レコードレイアウト設定。
15   */
16  public class RecordLayout {
17  
18      /** レコードタイプ。 */
19      @Getter
20      @Setter
21      private RecordType type;
22  
23      /** 当該レコードタイプだと判断する1文字目のコード値 ※0→Header、1→Data、8→Trailer、9→End etc。 */
24      @Getter
25      @Setter
26      private String codeValue;
27  
28      /** 項目設定リスト。 */
29      @Getter
30      @Setter
31      private List<ItemLayout> itemList;
32  
33      /** バイト長。 */
34      @Setter
35      private int byteLength;
36  
37      /** 比較キー項目リスト。 */
38      private List<String> compareKeyItemList;
39  
40      /** 全項目リスト。 */
41      private List<String> compareItemNameList;
42  
43      /** 比較条件マップ。 */
44      private Map<String, CompareCriteria> criteriaMap;
45  
46      /**
47       * 1レコードのバイト長を返します。
48       *
49       * @return 1レコードのバイト長
50       */
51      public int getByteLength() {
52          if (byteLength <= 0) {
53              for (final ItemLayout item : itemList) {
54                  byteLength = byteLength + item.getByteLength();
55              }
56          }
57          return byteLength;
58      }
59  
60      /**
61       * 比較キー項目をリストで返します。
62       *
63       * @return 比較キー項目リスト
64       */
65      public List<String> getCompareKeyItemList() {
66          if (compareKeyItemList == null) {
67              compareKeyItemList = new ArrayList<String>();
68              for (final ItemLayout item : itemList) {
69                  if (item.isCompareKey()) {
70                      compareKeyItemList.add(item.getId());
71                  }
72              }
73          }
74          return compareKeyItemList;
75      }
76  
77      /**
78       * 全項目をリストで返します。
79       *
80       * @return 全項目リスト
81       */
82      public List<String> getCompareItemNameList() {
83          if (compareItemNameList == null) {
84              compareItemNameList = new ArrayList<String>();
85              for (final ItemLayout item : itemList) {
86                  compareItemNameList.add(item.getId());
87              }
88          }
89          return compareItemNameList;
90      }
91  
92      /**
93       * 項目と対応する比較条件をMapで返します。
94       *
95       * @return 比較条件マップ
96       */
97      public Map<String, CompareCriteria> getCriteriaMap() {
98          if (criteriaMap == null) {
99              criteriaMap = new LinkedHashMap<String, CompareCriteria>();
100             for (final ItemLayout item : itemList) {
101                 criteriaMap.put(item.getId(), item.getCriteria());
102             }
103         }
104         return criteriaMap;
105     }
106 }
